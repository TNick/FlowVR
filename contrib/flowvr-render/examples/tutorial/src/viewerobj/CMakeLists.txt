SET(target viewerobj)

FIND_PACKAGE(OpenGL REQUIRED)
FIND_PACKAGE(GLUT REQUIRED)

INCLUDE_DIRECTORIES(${FLOWVR_base_INCLUDE_DIR}
                    ${FLOWVR_mod_INCLUDE_DIR}
                    ${FLOWVR_ftl_INCLUDE_DIR}
                    ${FLOWVRRENDER_INCLUDE_DIR}
                    ${OPENGL_INCLUDE_DIR} ${GLUT_INCLUDE_DIR})

LINK_LIBRARIES(${FLOWVR_base_LIBRARY} ${FLOWVR_mod_LIBRARY} ${FLOWVR_ftl_LIBRARY} ${FLOWVRRENDER_LIBRARY} ${CMAKE_THREAD_LIBS_INIT} ${OPENGL_LIBRARIES} ${GLUT_LIBRARIES})

# used only for out-of-source build
ADD_EXECUTABLE(${target} ${target}.cpp)
INSTALL(TARGETS ${target}  DESTINATION ${MODULE_DIR})
